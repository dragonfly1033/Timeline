from tkinter import *

def togglesidebar():

    global sidebar, sidebarPI, sidebarout, sideline, sidelinePI

    if(sidebarout):

        sidebarPI = sidebar.place_info()

        sidebar.place_forget()

        sidelinePI = sideline.place_info()

        sideline.place_forget()

    else:

        sidebar.place(sidebarPI)

        sidebar.lift(canvas)

        sideline.place(sidelinePI)

        sideline.lift(canvas)

        for i in range((130//2)+1):

            sidebar.update()
            sideline.update()
            sidebar.place(x=0, y=602-(i*2), anchor='nw')
            sideline.place(x=0, y=600-(i*2), anchor='nw')

    sidebarout = not sidebarout
